```go
func (m *UAManager) GetDiago(nodeID string) (*diago.Diago, error) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    managed, exists := m.agents[nodeID]
    if !exists {
        return nil, fmt.Errorf("UA not found for node %s", nodeID)
    }
    return managed.dg, nil
}
```
